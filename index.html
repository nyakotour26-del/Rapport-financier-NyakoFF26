// Charger les détails des dépenses - VERSION CORRIGÉE async function loadDetailsDepenses() { try { console.log('Chargement des détails des dépenses...'); const response = await fetch(DETAILS_DEPENSES_URL); if (response.ok) { const csvText = await response.text(); const lines = parseCSV(csvText); // Liste complète des catégories const categories = [ 'Admin', 'Artisanat', 'Cadeaux', 'Eglise', 'Fournitures et divers', 'Frais bancaire', 'Logistique', 'Repetition', 'Nourritures&snack', 'Tenue', 'Soutien participants', 'Voiture & frais voyage', 'Vol&frais voyage', 'Sante', 'Visa' ]; // Réinitialiser les données categories.forEach(cat => { detailsDepensesParCategorie[cat] = []; }); // Parcourir les lignes à partir de la ligne 3 (index 2) for (let i = 2; i <= 16 && i < lines.length; i++) { const row = lines[i]; if (row && row.length >= 7) { // MODIFICATION ICI : Colonne A (index 0) pour le libellé const libelle = row[0] ? row[0].trim() : ''; // Colonne B (index 1) pour la catégorie const categorie = row[1] ? row[1].trim() : ''; // Vérifier si c'est une catégorie valide const categorieTrouvee = categories.find(cat => cat.toLowerCase() === categorie.toLowerCase() ); if (categorieTrouvee && libelle && libelle !== '') { // Lire les colonnes C à G (index 2 à 6) const ariaryEntree = parseFloat(cleanNumber(row[2] || '0')); const ariarySortie = parseFloat(cleanNumber(row[3] || '0')); const euroEntree = parseFloat(cleanNumber(row[4] || '0')); const euroSortie = parseFloat(cleanNumber(row[5] || '0')); const banqueEuro = parseFloat(cleanNumber(row[6] || '0')); detailsDepensesParCategorie[categorieTrouvee].push({ 'libelle': libelle, 'Ariary Entrée': ariaryEntree, 'Ariary Sortie': ariarySortie, 'Euro Entrée': euroEntree, 'Euro Sortie': euroSortie, 'Banque Euro': banqueEuro }); } } } console.log('✅ Détails dépenses chargés:'); categories.forEach(cat => { if (detailsDepensesParCategorie[cat] && detailsDepensesParCategorie[cat].length > 0) { console.log(` - ${cat}: ${detailsDepensesParCategorie[cat].length} ligne(s)`); if (cat === 'Visa') { detailsDepensesParCategorie[cat].forEach((d, idx) => { console.log(` ${idx+1}. ${d.libelle}`); }); } } }); } else { console.error('Erreur chargement détails dépenses:', response.status); } } catch (error) { console.error('❌ Erreur chargement détails dépenses:', error); } }
